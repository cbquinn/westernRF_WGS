library(tidyverse)


dp <- read_delim("/group/ctbrowngrp2/cbquinn/fox4/0_bams/indelrealigned/qcmetrics/depthsummary_allfoxes_wheader.txt") %>%
	rename(sampleID=filename) %>%
	mutate(sampleID=gsub("_mapq30.mosdepth.summary.txt", "", sampleID)) %>%
	select(sampleID, q30_dp, q30_Xdp)

info <- read_tsv("/group/ctbrowngrp2/cbquinn/fox4/0_bams/samplelist.txt", col_names=c("RFID", "sampleID", "SID", "pop", "keep")) %>%
select(sampleID, pop, keep) %>%
mutate(pop=factor(pop, levels= c("SN","SNF1","LAS", "LAS2019", "ORC","WAC" , "RM",  "SV", "SVNN", "ECAN","VT", "AK","RUS"))) %>%
filter(keep == TRUE) %>%
arrange(pop, sampleID) %>%
select(-keep)

all <- left_join(info, dp) %>%
	mutate(q30_dp_round=round(q30_dp, digits=0), q30_Xdp_round=round(q30_Xdp, digits=0)) 

write_tsv(all, "/group/ctbrowngrp2/cbquinn/fox4/0_bams/indelrealigned/qcmetrics/depthsummary_34_wheaders.txt")

all %>%
	select(sampleID, q30_dp_round) %>%
	write_tsv("/group/ctbrowngrp2/cbquinn/fox4/0_bams/samplelist_34_wdepths.txt", col_names=FALSE)
